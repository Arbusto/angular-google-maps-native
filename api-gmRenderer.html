To set a panel to display the directions steps, use the gm-then like: 


.controller('MyCtrl', function ($scope) {
  $scope.setPanel = function (renderer) {
    renderer.setPanel(document.getElementById('panelId'));
  }
})

<gm-directions destination="destination" origin="origin" options="{travelMode: google.maps.DirectionsTravelMode.DRIVING}">
  <gm-renderer gm-then="setPanel(renderer)"></gm-renderer>
</gm-directions>
